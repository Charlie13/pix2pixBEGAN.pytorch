import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline
import scipy
from scipy import signal

def parse(LOG_FILE_PATH, LOG_FILE_NAME, SET_ID):
    log_filename = '%s/%s' % (LOG_FILE_PATH, LOG_FILE_NAME % SET_ID)
    print log_filename
    entries = [entry.strip().split('\t') \
               for entry in open(log_filename, 'r')]
    iter = []
    errD = []
    errG = []
    L_img = []
    measure = []
    M = []
    for entry in entries:
        iter.append(entry[0])
        errD.append(entry[1])
        errG.append(entry[2])
        L_img.append(entry[3])
        measure.append(entry[4])
        M.append(entry[5])
        
    errD = np.asarray(errD).astype(np.float)
    errG = np.asarray(errG).astype(np.float)
    iter = np.asarray(iter).astype(np.float)
    L_img = np.asarray(L_img).astype(np.float)
    measure = np.asarray(measure).astype(np.float)
    M = np.asarray(M).astype(np.float)
    
    errD = scipy.signal.medfilt(errD, 101)[:-51];
    errG = scipy.signal.medfilt(errG, 101)[:-51];
    measure = scipy.signal.medfilt(measure, 101)[:-51];
    
    return errD, errG, iter, L_img, M, measure
